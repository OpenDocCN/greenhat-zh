```
  2  * Check the flag.                              *
  3  ************************************************/
  4 #include <iostream>
  5
  6 int main()

  8     char ch;    // The flag
  9
 10     ch = 0xFF;  // Set the flag
 11
 12     // Check the flag
 13     if (ch == 0xFF)
 14         std::cout << "Success\n";
 15     else
 16         std::cout << "Fails\n"; 
 17
 18     return (0);
 19 }

```
